### 正则
let num = 1234567890;
let reg = /\d{1,3}(?=(\d{3})+$)/g;     //1233432543543636
String(num).replace(reg, '$&,'); //"1,234,567,890"

// 正则匹配方法二
let num = 1234567890;
let reg = /\B(?=(\d{3})+$)/g;   
String(num).replace(reg, ','); //"1,234,567,890"


1. ?= 表示正向引用
2. $& 表示与正则表达式相匹配的内容，可查看replace()
3. \B 非单词边界

### reduce
function format(num) {
  var str = num+'';
  return str.split("").reverse().reduce((prev, next, index) => {
    return ((index % 3) ? next : (next + ',')) + prev;
  })
}
let num = 1234567890; 
format(num);
